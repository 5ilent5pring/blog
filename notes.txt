--- geoCTI Pipeline (run from ~/5ilent5pring/geoCTI) ---

# Full auto: collect + analyze + publish to blog in one go
python main.py collect
python main.py analyze --publish

# Step by step (if you want to inspect before publishing)
python main.py collect                  # Scrape RSS feeds
python main.py collect --cti            # Only CTI feeds
python main.py collect --geo            # Only geopolitics feeds
python main.py collect --hybrid         # Only hybrid feeds
python main.py analyze                  # SAT analysis (no publish)
python main.py publish                  # Publish queued drafts to blog
python main.py review list              # List pending drafts in queue

# Utilities
python main.py status                   # Pipeline stats (inbox, queue, archive)
python main.py export                   # Export drafts as text files
python main.py test-llm                 # Test Ollama + model connection
python main.py cleanup                  # Clear inbox + queue
python main.py cleanup --all            # Clear everything including archive
python main.py cleanup --undo-last      # Re-fetch last batch of articles
python main.py kill-ollama              # Force-stop Ollama

# Override model for analysis
python main.py analyze --model gemma4:26b --publish
python main.py analyze --max-analyze 2 --publish   # Cap deep analysis count
